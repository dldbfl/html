입력받아 배열에 저장

function proc1(){
	arr = new Array();

	while(true){

	//입력
		input = prompt("이름입력");

	//배열에 저장
		if(input == ""||input == null )break;
		arr.push(input);
	}
	document.getElementById('result1').innerHTML = arr;
}
